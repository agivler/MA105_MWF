\question[6] Use the table of results from 1000 pre-employment drug screening tests to answer the questions below.

\begin{tabular} {|r|c|c|} \hline
& {\bfseries Positive Test Result} & {\bfseries Negative Test Result} \\ \hline
{\bfseries Subject uses drugs} & 99 & 8 \\ \hline
{\bfseries Subject does not use drugs} & 82 & 811\\ \hline
\end{tabular}

\begin{parts}
\part What is the probability of a person from this group having a positive test result?
\begin{solution}[\stretch{1}] $\frac{99+82}{99+82+8+811} = \frac{181}{1000}= 0.181$
\end{solution}
\part What is the probability of a person from this group being a drug user?
\begin{solution}[\stretch{1}] $\frac{99+8}{1000} = 0.107$
\end{solution}
\part What is the probability of a person from this group being a drug user or having a positive test result?
\begin{solution}[\stretch{1}] $\frac{99+8+82}{1000} = 0.189$
\end{solution}
\part What is the probability of a false positive?
\begin{solution}[\stretch{1}] $\frac{82}{1000} = 0.082$
\end{solution}
\part What is the probability of a person having a negative test result or not being a drug user?
\begin{solution}[\stretch{1}] $\frac{8+811+82}{1000} = 0.901$
\end{solution}
\part Given that a person doesn't use drugs, what is the probability that they have a positive test result?
\begin{solution}[\stretch{1}] $\frac{82}{82+811} = 0.092$
\end{solution}
\end{parts}

\question[9] Use the given table of results about people who were included in a survey to answer each of the questions below.

\begin{tabular}{|r|c|c|c|c|c|c|} \hline
 & \multicolumn{6}{c|}{{\bfseries Age}} \\ \hline
 & 18--21 & 22--29 & 30--39 & 40--49 & 50--59 & 60+ \\ \hline
{\bfseries Responded} & 73 & 255 & 245 & 136 & 138 & 202 \\ \hline
{\bfseries Refused} & 11 & 20 & 33 & 16 & 27 & 49\\ \hline
\end{tabular}

\begin{parts}
\part What is the probability that a selected person refused to answer?
\begin{solution}[\stretch{1}] $\frac{11+20+33+16+27+49}{11+20+33+16+27+49+73+255+245+136+138+202} = \frac{156}{1205} = 0.129$
\end{solution}
\part What is the probability that a selected person refused to answer or was 60 or older?
\begin{solution}[\stretch{1}] $\frac{156+202}{1205} = 0.297$
\end{solution}
\part What is the probability that a selected person was in the 18--21 age range?
\begin{solution}[\stretch{1}] $\frac{73+11}{1205} = 0.070$
\end{solution}
\part What is the probability that a selected person responded and was in the 22--29 age range?
\begin{solution}[\stretch{1}] $\frac{255}{1205} = 0.212$
\end{solution}
\part What is the probability that a selected person was younger than 22 or was older than 59?
\begin{solution}[\stretch{1}] $\frac{73+11+202+49}{1205} = 0.278$
\end{solution}
\part What is the probability that a selected person was younger than 22 or was older than 59 or refused to answer?
\begin{solution}[\stretch{1}] $\frac{73+11+202+49+20+33+16+27}{1205} = 0.358$
\end{solution}
\part What is the probability that a person was younger than 22 and responded?
\begin{solution}[\stretch{1}] $\frac{73}{1205} = 0.061$
\end{solution}
\part Given that a person is in the 30--39 age range, what is the probability that they refused to answer?
\begin{solution}[\stretch{1}] $\frac{33}{33+245} = 0.119$
\end{solution}
\part Given that a person responded, what is the probability that they are 60 or older?
\begin{solution}[\stretch{1}] $\frac{202}{73+255+245+136+138+202} = 0.193$
\end{solution}
\end{parts}
